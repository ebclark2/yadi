253 template <typename T>
254 struct function_call_via_yaml {
255  using result_type = function_traits_result_type<T>;
256  using params_type = function_traits_params_type<T>;
257  using function_type = function_traits_function_type<T>;
258  using index_sequence = function_traits_index_sequence<T>;
259 
260  static result_type call(function_type func, YAML::Node const& yaml);
261 
262  private:
263  function_call_via_yaml(function_type func, params_type* params) : func(std::move(func)), params(params) {}
264 
265  template <std::size_t... I>
266  result_type call_func(std::index_sequence<I...>) {
267  // return func(static_cast<typename std::tuple_element<I, params_type>::type
268  // &&>(std::get<I>(*params))...);
269  return func(std::move(std::get<I>(*params))...);
270  }
271 
272  result_type delayed_dispatch() { return call_func(index_sequence{}); }
273 
274  function_type func;
275  params_type* params;
276 };

406 template <typename BT>
407 initializer_type_t<BT> make_caching_initializer(initializer_type_t<BT> const& initializing_initializer) {
408  static_assert(std::is_same<ptr_type_t<BT>, std::shared_ptr<BT>>::value,
409  "ptr_type for BT must be shared_ptr to use caching initializer");
410 
411  using Cache = std::map<std::string, std::weak_ptr<BT>>;
412  Cache cache;
413  auto caching_initializer = [cache, initializing_initializer](YAML::Node const& config) mutable -> ptr_type_t<BT> {
414  // TODO Apply formatting to yaml_key
415  std::string yaml_key = YAML::Dump(config);
416  typename Cache::iterator cache_it = cache.find(yaml_key);
417  if (cache_it != cache.end()) {
418  ptr_type_t<BT> instance = cache_it->second.lock();
419  if (instance) {
420  return instance;
421  }
422  }
423 
424  ptr_type_t<BT> new_instance = initializing_initializer(config);
425  cache[yaml_key] = new_instance;
426  return new_instance;
427  };
428 
429  return caching_initializer;
430 }
