126 template <typename OT>
127 OT from_yaml(YAML::Node const& factory_config) {
128  using BT = meta::derive_base_type_t<OT>;
129  using DOT = OT; // derive_output_type_t<OT>;
130  if (adapter<OT, OT>::direct_from_yaml) {
131  return adapter<BT, DOT>::create(type_by_value_key(), factory_config);
132  }
133 
134  if (!factory_config.IsDefined()) {
135  throw std::runtime_error("Factory config not defined");
136  }
137 
138  if (factory_config.IsScalar()) {
139  std::string type = factory_config.as<std::string>("");
140  if (type.empty()) {
141  throw std::runtime_error("Factory config scalar not valid");
142  }
143 
144  return adapter<BT, DOT>::create(type);
145  }
146 
147  if (factory_config.IsMap()) {
148  YAML::Node typeNode = factory_config["type"];
149  if (!typeNode.IsDefined()) {
150  throw std::runtime_error("Factory config type not defined");
151  }
152  std::string type = typeNode.as<std::string>("");
153  if (type.empty()) {
154  throw std::runtime_error("Factory config type not valid");
155  }
156 
157  YAML::Node configNode = factory_config["config"];
158  return adapter<BT, DOT>::create(type, configNode);
159  }
160 
161  throw std::runtime_error("Factory config not valid, YAML must be scalar string or map");
162 }

164 template <typename OT, typename OI>
165 void from_yamls(YAML::Node const& factory_configs, OI out) {
166  if (!factory_configs.IsDefined()) {
167  throw std::runtime_error("From YAML factory configs not defined");
168  }
169  // If it's not a sequence then parse single
170  if (!factory_configs.IsSequence()) {
171  *out = from_yaml<OT>(factory_configs);
172  ++out;
173  return;
174  }
175 
176  // A sequence!
177  for (YAML::Node const& entry : factory_configs) {
178  *out = from_yaml<OT>(entry);
179  ++out;
180  }
181 }
